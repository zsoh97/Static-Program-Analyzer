    a = 1;
    if (b == 2) then {
        c = 3;
        if (d == 4) then {
            e = 5;
        } else {
            f = 6;
        }
        g = 7;
    } else {
        h = 8;
    }

    j = 10;
    if (k == 11) then {
        l = 12;
        while (m == 13) {
            n = 14;
        }
        o = 15;
    } else {
        p = 16;
    }

}
